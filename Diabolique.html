#' Movie Data Scraper
#' @param tmdb_api_key Your TMDB API key
scrape_movie_data <- function(tmdb_api_key) {
  
  # Initialize empty lists for storing data
  reviews <- list()
  metadata <- list()
  
  # Letterboxd scraping function
  scrape_letterboxd <- function() {
    tryCatch({
      # Set URL for Les Diaboliques (1955)
      url <- "https://letterboxd.com/film/diabolique-1955/"
      
      # Read the webpage
      page <- read_html(url)
      
      # Get overall rating
      rating <- page %>%
        html_node("meta[name='twitter:data2']") %>%
        html_attr("content") %>%
        str_extract("\\d+\\.\\d+") %>%
        as.numeric()
      
      # Get reviews
      review_texts <- page %>%
        html_nodes(".review-text") %>%
        html_text() %>%
        str_trim()
      
      # Create reviews dataframe
      letterboxd_reviews <- tibble(
        source = "Letterboxd",
        text = review_texts,
        date = Sys.Date()  # You might want to extract actual review dates
      )
      
      # Update our data collections
      reviews[[length(reviews) + 1]] <- letterboxd_reviews
      metadata$letterboxd_rating <- rating
      
    }, error = function(e) {
      message("Error scraping Letterboxd: ", e$message)
    })
  }
  
  # TMDB API scraping function
  scrape_tmdb <- function(api_key, movie_id = "11661") {
    tryCatch({
      # Base URL for TMDB API
      base_url <- paste0("https://api.themoviedb.org/3/movie/", movie_id)
      
      # Get movie details
      movie_response <- GET(
        url = base_url,
        query = list(
          api_key = api_key,
          language = "en-US"
        )
      )
      
      movie_data <- fromJSON(rawToChar(movie_response$content))
      
      # Get reviews
      reviews_response <- GET(
        url = paste0(base_url, "/reviews"),
        query = list(
          api_key = api_key,
          language = "en-US"
        )
      )
      
      reviews_data <- fromJSON(rawToChar(reviews_response$content))
      
      # Create reviews dataframe
      if (length(reviews_data$results) > 0) {
        tmdb_reviews <- tibble(
          source = "TMDB",
          text = reviews_data$results$content,
          date = as.Date(substr(reviews_data$results$created_at, 1, 10))
        )
        reviews[[length(reviews) + 1]] <- tmdb_reviews
      }
      
      # Update metadata
      metadata$tmdb_rating <- movie_data$vote_average
      metadata$vote_count <- movie_data$vote_count
      metadata$popularity <- movie_data$popularity
      
    }, error = function(e) {
      message("Error scraping TMDB: ", e$message)
    })
  }
  
  # Execute scraping
  scrape_letterboxd()
  scrape_tmdb(tmdb_api_key)
  
  # Combine all reviews
  all_reviews <- bind_rows(reviews)
  
  # Create metadata dataframe
  metadata_df <- as_tibble(metadata)
  
  # Export data
  write_csv(all_reviews, "diabolique_1955_reviews.csv")
  write_csv(metadata_df, "diabolique_1955_metadata.csv")
  
  # Return results as a list
  return(list(
    reviews = all_reviews,
    metadata = metadata_df
  ))
}

# Example usage:
